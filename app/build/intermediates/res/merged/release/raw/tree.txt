function grow() {
	var whichTree = Math.floor(Math.random()*100);
	if (treeArray[whichTree]==0) {
		treeArray[whichTree] = 1;
		displayIt(whichTree, 1);
	}
	else {
		var displayed = 0;
		for (i=treeArray+1; i<100; i++) {

			if (treeArray[i]==0) {
				treeArray[i] = 1;
				displayIt(i, 1);
				displayed = 1;
				break;
			}
		}
		if (displayed==0) {
			for (i=0; i<100; i++) {
				if (treeArray[i]==0) {
					treeArray[i] = 1;
					displayIt(i, 1);
				}
				break;
			}
		}
	}
}
